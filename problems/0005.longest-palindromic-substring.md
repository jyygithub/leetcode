> [https://leetcode.cn/problems/longest-palindromic-substring/](https://leetcode.cn/problems/longest-palindromic-substring/)

**Description**
```text
Given a string s, return the longest palindromic substring in s.
```

**Example 1:**
```text
Input: s = "babad"
Output: "bab"
Explanation: "aba" is also a valid answer.
```
**Example 2:**
```text
Input: s = "cbbd"
Output: "bb"
```
**Example 3:**
```text
Input: nums = [3,3], target = 6
Output: [0,1]
```

**Solution**
```kotlin
class Solution {
    fun longestPalindrome(s: String): String {
        val len = s.length
        if (len < 2) return s
        var maxLen = 1
        var begin = 0
        val dp = Array(len) { BooleanArray(len) }
        for (i in 0 until len) {
            dp[i][i] = true
        }
        val charArray = s.toCharArray()
        for (i in 2..len) {
            for (j in 0 until len) {
                val k = i + j - 1
                if (k >= len) break
                if (charArray[j] != charArray[k]) dp[j][k] = false
                else if (k - j < 3) dp[j][k] = true
                else dp[j][k] = dp[j + 1][k - 1]

                if (dp[j][k] && k - j + 1 > maxLen) {
                    maxLen = k - j + 1
                    begin = j
                }
            }
        }
        return s.substring(begin, begin + maxLen)
    }
}
```
